<PropertyTable>
  <PropertyTable.Row
    property="grid.showHorizontalLines"
    type="boolean"
    description="determines whether or not to show horizontal grid lines"
    chartsAffected={['BarChart', 'LineChart', 'MultiSeriesBarChart']}
  />
  <PropertyTable.Row
    property="grid.horizontalOverflow"
    type="boolean"
    description="determines whether or not to extend the grid horizontally to the edge of the chart"
    chartsAffected={['BarChart', 'LineChart', 'MultiSeriesBarChart']}
  />
  <PropertyTable.Row
    property="grid.verticalOverflow"
    type="boolean"
    description="determines whether or not to allow vertical overflow (the top most horizontal grid line is hidden). Even if enabled, there will only be horizontal overflow in certain scenarios."
    chartsAffected={['BarChart', 'LineChart', 'StackedAreaChart']}
  />
  <PropertyTable.Row
    property="grid.color"
    type="string"
    description="a CSS color string that gets applied to the grid lines"
    chartsAffected={[
      'BarChart',
      'LineChart',
      'MultiSeriesBarChart',
      'StackedAreaChart',
    ]}
  />
  <PropertyTable.Row
    property="grid.horizontalMargin"
    type="number"
    description="the number of pixels between the left and right edges of the grid and the edges of the chart container"
    chartsAffected={[
      'BarChart',
      'LineChart',
      'MultiSeriesBarChart',
      'StackedAreaChart',
    ]}
  />
</PropertyTable>
