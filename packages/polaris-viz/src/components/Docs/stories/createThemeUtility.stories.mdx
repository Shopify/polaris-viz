<p>
  The library also exports a <code>createTheme</code> utility function that you
  can use to create multiple themes avoiding duplication:
</p>

```tsx
export const createTheme = (
  theme: PartialTheme,
  baseTheme: Theme = PolarisVizTheme,
): Theme => {
  //...This function creates a complete theme from the
  // partial theme provided. It will overwrite the base theme
  // with the properties of the partial theme
};
```

<p>
  If you call the function passing only a partial theme, it will use the
  <code>PolarisVizDefaultTheme</code> as a base theme:
</p>
<p>
  <code>myDefaultTheme</code> has all the properties from{' '}
  <code>PolarisVizDefaultTheme</code> , but has red background and square
  corners
</p>

```tsx
import {createTheme} from '@shopify/polaris-viz';

const myDefaultTheme = createTheme({
  barTheme: {
    hasRoundedCorners: false,
  },
  chartContainer: {
    backgroundColor: 'red',
  },
});
```

<p>
  You can also pass your own theme as the base theme.
  <code>myOtherTheme</code> will inherit all the properties of myDefaultTheme but
  will use green background
</p>

```tsx
import {createTheme} from '@shopify/polaris-viz';

const myOtherTheme = createTheme(
  {
    chartContainer: {
      backgroundColor: 'green',
    },
  },
  myDefaultTheme,
);
```
